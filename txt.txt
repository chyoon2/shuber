-Take the 1st split it into an array.

function isPresent (array) { 

  string1 = array[0];               // string1 = "i"
  array1 = string1.split("");       // array1 = [ 'i']
  array2 = array[1].split("");      // array2 = [ 'i', 's']

  foreach( let element of array2) {     //element: s
    let counter = 0;                    //counter: 0
    for(i = 0; i < array[0].length; i++) {     // i: 0
      if( string1[i] == element) {       //
        counter++
      }
    }  
    if (counter == 0){
      return false
    } 
  } 
  return true
}
